    𑌸𑍁𑌨𑍍𑌦𑌰𑌜𑌾𑌮𑌾𑌤𑍃𑌮𑍁𑌨𑍇𑌃 𑌪𑍍𑌰𑌪𑌦𑍍𑌯𑍇 𑌚𑌰𑌣𑌾𑌮𑍍𑌬𑍁𑌜𑌮𑍍 ।
    𑌸𑌂𑌸𑌾𑌰𑌾𑌰𑍍𑌣𑌵𑌸𑌮𑍍𑌮𑌗𑍍𑌨𑌜𑌨𑍍𑌤𑍁𑌸𑌨𑍍𑌤𑌾𑌰𑌪𑍋𑌤𑌕𑌮𑍍 ॥

𑌶𑍍𑌰𑍀𑌵𑌾𑌦𑌿𑌕𑍇𑌸𑌰𑌿𑌵𑌿𑌰𑌚𑌿𑌤𑌮𑍍 𑌅𑌷𑍍𑌟𑌕𑌂 𑌵𑍍𑌯𑌾𑌖𑍍𑌯𑌾𑌤𑍁𑌂 𑌯𑌤𑍇 । 𑌇𑌨𑍍𑌦𑍁𑌵𑌦𑌨𑌾𑌵𑍃𑌤𑍍𑌤𑍇𑌨 𑌲𑌿𑌖𑌿𑌤𑌮𑍍 । 𑌇𑌨𑍍𑌦𑍁𑌵𑌦𑌨𑌾 𑌭𑌜𑌸𑌨𑍈𑌃 𑌸𑌗𑍁𑌰𑍁𑌯𑍁𑌗𑍍𑌮𑍈𑌃 ।

    𑌶𑍍𑌰𑍀𑌮𑌦𑌕𑌲𑌙𑍍𑌕𑌪𑌰𑌿𑌪𑍂𑌰𑍍𑌣 𑌶𑌶𑌿𑌕𑍋𑌟𑌿𑌶𑍍𑌰𑍀𑌧𑌰 𑌮𑌨𑍋𑌹𑌰 𑌸𑌟𑌾𑌪𑌟𑌲𑌕𑌾𑌨𑍍𑌤 ।
    𑌪𑌾𑌲𑌯 𑌕𑍃𑌪𑌾𑌲𑌯 𑌭𑌵𑌾𑌮𑍍𑌬𑍁𑌧𑌿𑌨𑌿𑌮𑌗𑍍𑌨𑌂 𑌦𑍈𑌤𑍍𑌯𑌵𑌰𑌕𑌾𑌲 𑌨𑌰𑌸𑌿𑌂𑌹 𑌨𑌰𑌸𑌿𑌂𑌹 ॥ ௧ ॥
